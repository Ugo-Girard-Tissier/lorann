9   /**
10   * <h1> the class Elements. <h1>
11   * <p>
12   * this class is use to create every elements of the game like bone, vertical bone and so on
13   * </p>
14   * <p>
15   * the Elements class is a public class with 2 public attributes and 6 methods
16   * </p>
17   * <p>
18   * the Elements class implement ISquare because he need to be show on a IBoard
19   * </p>
20   * @author Sulyven, Enzo, Abdel, Arthur and Ugo group
21   * @version 1.0
22   * @see Elements
23   * @see getImage
24   * @see loadImage
25   * @see setImage
26   * @see getImageName
27   * @see setImageName
28   *
29   */
30  
31  public class Elements implements ISquare
32  {
33  	/**Image of an elements */
34  	public Image img_ob;
35  	/**Name the file which contain the sprite needed*/ 
36  	public String nameFile;
37  	
38  	/**This constructor is used to set a sprite for each elements of the game (bone, lorann, and so on*/ 
39  	public Elements(final String nameFile)
40  	{
41  		this.setImageName(nameFile);
42  	}
43  	
44  	/** Override method use to get the sprite of an element*/
45  	@Override
46      public final Image getImage() 
47  	{
48          return this.img_ob;
49      }
50  
51      /* Load image.
52       *
53       * @throws IOException Signals that an I/O exception has occurred.
54       */
55      public final void loadImage() throws IOException 
56      {
57          this.setImage(ImageIO.read(new File("annex/" + this.getImageName())));
58      }
59  
60      /* Sets the image.
61      *
62       * @param image the new image
63       */
64      protected void setImage(final Image image) 
65      {
66          this.img_ob = image;
67      }
68  
69      /* Gets the image name.
70       *
71       * @return the image name
72       */
73      protected String getImageName() 
74      {
75          return this.nameFile;
76      }
77  
78      /* Sets the image name.
79       *
80       * @param imageName the imageName to set
81       */
82      protected void setImageName(final String imageName) {
83          this.nameFile = imageName;
84      }
85  }

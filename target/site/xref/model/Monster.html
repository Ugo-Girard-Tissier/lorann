50  	/** This constructor is used to set the start position of a monsters (random position) */
51  	public Monster(String nameFile, map Map) 
52  	{
53  		super(nameFile);
54  		this.map = Map;
55  		this.RandomStartPosition();
56  		this.setPosition(getStartX(), getStartY());
57  	}

59  	/** Generate random x and y position (start position of the Monster) */
60  	public final void RandomStartPosition()
61  	{
62  		int valuesX = 0;
63  		int valuesY = 0;
64  		while (map.mapRead[valuesX][valuesY] != 'N')
65  		{
66  			Random rX = new Random();
67  			Random rY = new Random();
68  			valuesX = 0 + rX.nextInt(19 - 0);
69  			valuesY = 0 + rY.nextInt(11 - 0);
70  		}
71  		this.startX = valuesX;
72  		this.startY = valuesY;
73  	}
74  	
75  	/** this method reset the position of a monster (into a random one) */
76  	public final void resetMonster()
77  	{
78  		this.RandomStartPosition();
79  		this.setPosition(getStartX(), getStartY());
80  		this.aliveMonster = true;
81  	}

83  	/** this method allow a monster to move randomly (9 direction, a direction is choose randomly) */
84  	public final void RamdomMovingMonster()
85  	{
86  		Random Mouvement = new Random();
87  		int mouvementDirection = 0 + Mouvement.nextInt(9 - 0);
88  		switch(mouvementDirection)
89  		{
90  			case 1:
91  				this.rightMouvementMonster();
92  				break;
93  			case 2:
94  				this.leftMouvementMonster();
95  				break;
96  			case 3:
97  				this.forwardMouvementMonster();
98  				break;
99  			case 4:
100 				this.backwardMouvementMonster();
101 				break;
102 			case 5:
103 				this.diagonalLeftForwardMouvementMonster();
104 				break;
105 			case 6:
106 				this.diagonalRightForwardMouvementMonster();
107 				break;
108 			case 7:
109 				this.diagonalLeftBackwardMouvementMonster();
110 				break;
111 			case 8:
112 				this.diagonalRightBackwardMouvementMonster();
113 				break;
114 		}
115 		
116 	}

118 	/** this method allow a monster to stalk Lorann only in the x line (width) and y line (height) */
119 	public final void monstersTrackingLorann()
120 	{
121 		if (this.getX() < this.map.getLorann().getX())
122 			this.rightMouvementMonster();
123 		else if(this.getX() > this.map.getLorann().getX())
124 			this.leftMouvementMonster();
125 		else if(this.getY() < this.map.getLorann().getY())
126 			this.backwardMouvementMonster();
127 		else if(this.getY() > this.map.getLorann().getY())
128 			this.forwardMouvementMonster();
129 	}

152 	/** Right movement of the Monster, set the direction and add 1 to the x position */
153 	public final void rightMouvementMonster()
154 	{
155 		int NewX = this.getX();
156 		int NewY = this.getY();
157 		NewX = NewX + 1;
158 		if (this.map.mapRead[NewX][NewY] != 'B' 
159 				&& this.map.mapRead[NewX][NewY] != 'H' 
160 				&& this.map.mapRead[NewX][NewY] != 'V' 
161 				&& this.map.mapRead[NewX][NewY] != 'C' 
162 				&& this.map.mapRead[NewX][NewY] != 'G' 
163 				&& this.map.mapRead[NewX][NewY] != 'P' 
164 				&& this.map.mapRead[NewX][NewY] != 'O')
165 		{
166 			this.setPosition(NewX, NewY);
167 		}
168 	}
